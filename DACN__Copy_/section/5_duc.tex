\begin{figure}[H]
    \centering

    \caption{Activity Diagram mô tả luồng quản lý farm}

\noindent Biểu đồ mô tả quy trình tương tác giữa Quản trị viên và Hệ thống thông qua 5 chức năng chính: Thêm, Sửa, Xóa, Xem chi tiết và Nhập/Xuất dữ liệu. Quy trình chú trọng bước kiểm tra tính hợp lệ của dữ liệu trước khi lưu và luôn điều hướng quay lại danh sách trang trại sau mỗi tác vụ để tối ưu hóa việc quản lý liên tục.


\subsubsubsection{Quy trình quản lý thiết bị:} 
\begin{figure}[H]
    \centering
    \includegraphics[width=0.95\textwidth]{img/deviceact.jpg}
    \caption{Activity Diagram mô tả luồng quản lý thiệt bị}
    \label{fig:activity_analytics}
\end{figure}
\noindent Biểu đồ hoạt động Quản lý thiết bị: Mô tả luồng tương tác giữa Quản trị viên và Hệ thống. Quy trình bắt đầu từ việc truy cập danh sách, cho phép Admin thực hiện các nghiệp vụ cốt lõi như Thêm, Sửa, Xóa và Nhập/Xuất dữ liệu thiết bị. Ngoài ra, biểu đồ còn chi tiết hóa các thao tác quản lý Cảm biến (Sensors) bên trong từng thiết bị cụ thể. Hệ thống đóng vai trò phản hồi giao diện, xác nhận lưu trữ và cập nhật trạng thái sau mỗi hành động của người dùng.
